#!/usr/bin/env python
import sys

s = file('c2.in').read()
count={}
for c in s:
	if c not in count:
		count[c]=1
	else:
		count[c]+=1

m=min(count.values())
for c in s:
	if count[c] == m:
		sys.stdout.write(c)
print

import re

s = file('c3.in').read()
result = ''
for word in re.split('[a-z][A-Z]{3}([a-z])[A-Z]{3}[a-z]', s):
	if len(word) == 1:
		result+=word

#!/usr/bin/env python
import urllib, zipfile, re, StringIO
url='http://www.pythonchallenge.com/pc/def/channel.zip'

z = zipfile.ZipFile(StringIO.StringIO(urllib.urlopen(url).read()), 'r')
match = re.search('hint1: start from (\d+)', z.read('readme.txt'))
result = ''
while match:
	name = match.groups()[0]+'.txt'
	c = z.getinfo(name).comment.lower()
	if ord(c) in range(ord('a'), ord('z')) and result.find(c) < 0: result += c
	match = re.search('Next nothing is (\d+)', z.read(name))
print result
